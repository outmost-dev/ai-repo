API Integration (admin/src/utils/apiService.tsx:597):
```typescript
export const handleLogin = async (body: unknown) => {
  const headers = new Headers();
  headers.append('Content-Type', 'application/json');
  headers.append('Accept', 'application/json');
  headers.append('Authorization', 'Bearer ' + token);

  const response = await fetch(SIGNIN_API, {
    method: 'POST',
    headers,
    body: body ? JSON.stringify(body) : undefined,
  });
  const data = await response.json();

  if ([200, 201].includes(response?.status)) {
    return { status: response.status, data } as ResponseT;
  }

  if (response?.status === 401 && response.statusText === 'Unauthorized') {
    // Refresh the token
    localStorage.removeItem('authUser');
    window.location.replace('/auth/signin');
  }

  message.open({
    type: 'error',
    content: data?.message || response.statusText,
  });
  return { status: response.status, error: data } as unknown as ResponseT;
};
```

API Endpoint:
- URL: POST /v1/auth/admin-signin
- Request Body: { email: string, password: string }
- Success Response (201): { id, email, accessToken, refreshToken, username, role, fullName }
- Error Response (401/400): { message: string }

Pinia Store (stores/auth.ts):
```typescript
import { defineStore } from 'pinia';
import { ref } from 'vue';

export interface AuthUser {
  id: number;
  email: string;
  accessToken: string;
  refreshToken: string;
  username: string;
  role: number;
  fullName: string;
}

export const useAuthStore = defineStore('auth', () => {
  const authUser = ref<AuthUser | null>(null);
  const countUsers = ref(0);

  // Initialize from localStorage
  const storedUser = localStorage.getItem('authUser');
  if (storedUser) {
    authUser.value = JSON.parse(storedUser);
  }

  const storedCount = localStorage.getItem('countUsers');
  if (storedCount) {
    countUsers.value = JSON.parse(storedCount);
  }

  const login = (userData: AuthUser) => {
    authUser.value = userData;
    localStorage.setItem('authUser', JSON.stringify(userData));
  };

  const logout = () => {
    authUser.value = null;
    localStorage.setItem('authUser', '');
  };

  const setUserCount = (count: number) => {
    countUsers.value = count;
    localStorage.setItem('countUsers', JSON.stringify(count));
  };

  return {
    authUser,
    countUsers,
    login,
    logout,
    setUserCount,
  };
});
```

import { defineStore } from 'pinia';
import { ref, computed } from 'vue';

export interface AuthUser {
  id: number;
  email: string;
  accessToken: string;
  refreshToken: string;
  username: string;
  role: number;
  fullName: string;
}

export const useAuthStore = defineStore('auth', () => {
  // State
  const authUser = ref<AuthUser | null>(null);
  const countUsers = ref(0);

  // Initialize from localStorage
  const initializeAuth = () => {
    const storedUser = localStorage.getItem('authUser');
    if (storedUser) {
      try {
        authUser.value = JSON.parse(storedUser);
      } catch (error) {
        console.error('Failed to parse authUser from localStorage:', error);
        authUser.value = null;
      }
    }

    const storedCount = localStorage.getItem('countUsers');
    if (storedCount) {
      try {
        countUsers.value = JSON.parse(storedCount);
      } catch (error) {
        console.error('Failed to parse countUsers from localStorage:', error);
        countUsers.value = 0;
      }
    }
  };

  // Computed
  const isAuthenticated = computed(() => authUser.value !== null);
  const userRole = computed(() => authUser.value?.role ?? 0);
  const userName = computed(() => authUser.value?.fullName ?? '');

  // Actions
  const login = (userData: AuthUser) => {
    authUser.value = userData;
    localStorage.setItem('authUser', JSON.stringify(userData));
  };

  const logout = () => {
    authUser.value = null;
    localStorage.removeItem('authUser');
    localStorage.removeItem('countUsers');
  };

  const setUserCount = (count: number) => {
    countUsers.value = count;
    localStorage.setItem('countUsers', JSON.stringify(count));
  };

  const updateAccessToken = (token: string) => {
    if (authUser.value) {
      authUser.value.accessToken = token;
      localStorage.setItem('authUser', JSON.stringify(authUser.value));
    }
  };

  // Initialize on store creation
  initializeAuth();

  return {
    // State
    authUser,
    countUsers,

    // Computed
    isAuthenticated,
    userRole,
    userName,

    // Actions
    login,
    logout,
    setUserCount,
    updateAccessToken,
  };
});

export const handleLogin = async (body: unknown) => {
  try {
    const headers = new Headers();
    headers.append('Content-Type', 'application/json');
    headers.append('Accept', 'application/json');
    headers.append('Authorization', 'Bearer ' + token);

    const response = await fetch(SIGNIN_API, {
      method: 'POST',
      headers,
      body: body ? JSON.stringify(body) : undefined,
    });
    const data = await response.json();

    if ([200, 201].includes(response?.status)) {
      return { status: response.status, data } as ResponseT;
    }

    if (response?.status === 401 && response.statusText === 'Unauthorized') {
      localStorage.removeItem('authUser');
      window.location.replace('/auth/signin');
    }

    message.open({
      type: 'error',
      content: data?.message || response.statusText,
    });
    return { status: response.status, error: data } as unknown as ResponseT;
  } catch (error) {
    return error;
  }

import axios, { AxiosError } from 'axios';
import { message } from 'ant-design-vue';
import router from '@/router';

const ROOT_API =
  import.meta.env.VITE_ENV === 'production'
    ? 'https://api.somaway.ro/api'
    : import.meta.env.VITE_ENV === 'development'
    ? 'https://dev-api.somaway.ro/api'
    : 'http://localhost:3005/api';

export const SIGNIN_API = `${ROOT_API}/v1/auth/admin-signin`;

export interface ResponseT {
  status: number;
  data?: any;
  error?: any;
}

export interface LoginRequest {
  email: string;
  password: string;
}

export const handleLogin = async (body: LoginRequest): Promise<ResponseT> => {
  try {
    const response = await axios.post(SIGNIN_API, body, {
      headers: {
        'Content-Type': 'application/json',
        Accept: 'application/json',
      },
    });

    if ([200, 201].includes(response.status)) {
      return {
        status: response.status,
        data: response.data,
      };
    }

    return {
      status: response.status,
      error: response.data,
    };
  } catch (error) {
    if (axios.isAxiosError(error)) {
      const axiosError = error as AxiosError;

      // Handle 401 Unauthorized
      if (axiosError.response?.status === 401) {
        localStorage.removeItem('authUser');
        localStorage.removeItem('countUsers');
        router.push('/auth/signin');

        message.error('Sesiune expirată. Vă rugăm să vă autentificați din nou.');

        return {
          status: 401,
          error: axiosError.response.data,
        };
      }

      // Handle other HTTP errors
      const errorMessage =
        (axiosError.response?.data as any)?.message ||
        axiosError.message ||
        'A apărut o eroare la autentificare';

      message.error(errorMessage);

      return {
        status: axiosError.response?.status || 500,
        error: axiosError.response?.data,
      };
    }

    // Handle non-Axios errors
    console.error('Login error:', error);
    message.error('A apărut o eroare de rețea');

    return {
      status: 500,
      error: error,
    };
  }
};
