4. Redirect este instant (no loading screen)

Features:

- No loading UI (instant redirect)
- Constante exportate pentru subscription types
- Public page (no authentication required)

- Page: web/src/pages/index.tsx:1-16

```tsx
// web/src/components/pages/home/BenefitsList/index.tsx
class ErrorBoundary extends React.Component<
  { children: React.ReactNode },
  { hasError: boolean }
> {
  constructor(props: { children: React.ReactNode }) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError() {
    return { hasError: true };
  }

  componentDidCatch(error: Error, info: React.ErrorInfo) {
    console.error("ErrorBoundary caught an error", error, info);
  }

  render() {
    if (this.state.hasError) {
      return <h2>Something went wrong.</h2>;
    }
    return this.props.children;
  }
}

type BenefitsListProps = {
  experiences: string[];
  title?: React.ReactNode;
  description?: React.ReactNode;
  sectionName?: string;
};

const BenefitsList: React.FC<BenefitsListProps> = ({
  experiences,
  title,
  description,
  sectionName,
}) => {
  return (
    <section className={sectionName}>
      <div className="container">
        <Title level={2}>{title}</Title>
        <ErrorBoundary>
          <List
            className="experience-list"
            dataSource={experiences}
            renderItem={(item, index) => (
              <List.Item className={`experience-item experience-item-${index + 1}`}>
                <div dangerouslySetInnerHTML={{ __html: item }} />
              </List.Item>
            )}
          />
        </ErrorBoundary>
        {description && (
          <div className="experience-description">{description}</div>
        )}
      </div>
    </section>
  );
};
```

```tsx
// web/src/components/pages/home/Reviews/index.tsx
const reviews: string[] = [
  "/imgs/reviews/1.webp",
  "/imgs/reviews/2.webp",
  "/imgs/reviews/3.webp",
  "/imgs/reviews/4.webp",
  "/imgs/reviews/5.webp",
  "/imgs/reviews/6.webp",
  "/imgs/reviews/7.webp",
  "/imgs/reviews/8.webp",
  "/imgs/reviews/9.webp",
];

const Stories: React.FC = () => {
  const containerRef = useRef<HTMLDivElement>(null);
  const innerContainerRef = useRef<HTMLDivElement>(null);
  const firstElementRef = useRef<HTMLDivElement>(null);
  const [lightboxIndex, setLightboxIndex] = useState<number | null>(null);
  const [expanded, setExpanded] = useState<boolean>(false);

  const toggleExpand = () => {
    if (containerRef.current) {
      containerRef.current.classList.toggle("expanded");
    }
    if (expanded && containerRef.current) {
      window.scroll({
        top: containerRef.current.offsetTop - 250,
        behavior: "smooth",
      });
    }
    setExpanded(!expanded);
  };

  const openLightbox = (index: number) => {
    setLightboxIndex(index);
  };

  const closeLightbox = () => {
    setLightboxIndex(null);
  };

  return (
    <section className="reviews">
      <div className="container">
        <Title level={2}>Următoarea poveste poate fi a ta:</Title>

        <div
          className="reviews-container"
          ref={containerRef}
          style={{
            maxHeight: expanded
              ? innerContainerRef?.current?.offsetHeight
              : firstElementRef.current?.offsetHeight,
          }}
        >
          <div className="inner-container" ref={innerContainerRef}>
            {reviews.map((image, index) => (
              <div
                key={index}
                className="review-item"
                ref={index === 0 ? firstElementRef : null}
              >
                <Image
                  src={image}
                  alt={`Review ${index + 1}`}
                  onClick={() => openLightbox(index)}
                  layout="responsive"
                  width={500}
                  height={500}
                />
              </div>
            ))}
          </div>
        </div>

        <CustomButton type="link" size="small" onClick={toggleExpand}>
          {!expanded ? "Vezi mai multe testimoniale" : "Vezi mai putine testimoniale"}
        </CustomButton>

        <CustomButton type="primary" size="small" href="/home#vreau-abonament">
          Vreau în Somaway!
        </CustomButton>

        {lightboxIndex !== null && (
          <div className="lightbox" onClick={closeLightbox}>
            <Button className="close-button" onClick={closeLightbox}>
              X
            </Button>
            <Image
              src={reviews[lightboxIndex]}
              alt={`Review ${lightboxIndex + 1}`}
              layout="responsive"
              width={500}
            />
            <div className="navigation">
              <Button
                onClick={(e) => {
                  e.stopPropagation();
                  setLightboxIndex((prev) =>
                    prev !== null && prev > 0 ? prev - 1 : reviews.length - 1
                  );
                }}
              >
                Prev
              </Button>
              <Button
                onClick={(e) => {
                  e.stopPropagation();
                  setLightboxIndex((prev) =>
                    prev !== null && prev < reviews.length - 1 ? prev + 1 : 0
                  );
                }}
              >
                Next
              </Button>
            </div>
          </div>
        )}
      </div>
    </section>
  );
};
```

```vue
import { ref, computed } from 'vue';

const reviews: string[] = [
  "/imgs/reviews/1.webp",
  "/imgs/reviews/2.webp",
  "/imgs/reviews/3.webp",
  "/imgs/reviews/4.webp",
  "/imgs/reviews/5.webp",
  "/imgs/reviews/6.webp",
  "/imgs/reviews/7.webp",
  "/imgs/reviews/8.webp",
  "/imgs/reviews/9.webp",
];

const containerRef = ref<HTMLDivElement | null>(null);
const innerContainerRef = ref<HTMLDivElement | null>(null);
const firstElementRef = ref<HTMLDivElement | null>(null);
const lightboxIndex = ref<number | null>(null);
const expanded = ref<boolean>(false);

const containerStyle = computed(() => ({
  maxHeight: expanded.value
    ? `${innerContainerRef.value?.offsetHeight}px`
    : `${firstElementRef.value?.offsetHeight}px`,
}));

const toggleExpand = () => {
  if (containerRef.value) {
    containerRef.value.classList.toggle('expanded');
  }
  if (expanded.value && containerRef.value) {
    window.scroll({
      top: containerRef.value.offsetTop - 250,
      behavior: 'smooth',
    });
  }
  expanded.value = !expanded.value;
};

const openLightbox = (index: number) => {
  lightboxIndex.value = index;
};

const closeLightbox = () => {
  lightboxIndex.value = null;
};

const navigatePrev = (e: Event) => {
  e.stopPropagation();
  if (lightboxIndex.value !== null) {
    lightboxIndex.value = lightboxIndex.value > 0 ? lightboxIndex.value - 1 : reviews.length - 1;
  }
};

const navigateNext = (e: Event) => {
  e.stopPropagation();
  if (lightboxIndex.value !== null) {
    lightboxIndex.value = lightboxIndex.value < reviews.length - 1 ? lightboxIndex.value + 1 : 0;
  }
};
```
