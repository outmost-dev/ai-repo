```tsx
// web/src/layouts/AppLayout.tsx
import { Layout } from "antd/lib";
import React, { useEffect, useState } from "react";
import { Header } from "@/components/global/Header";
import { RouteHandler } from "@/middleware/RouteHandler";
import { FooterBlock } from "@/components/global/Footer/Footer";
import { useZoomAccess } from "@/hooks/use-zoom-access";

interface Props {
  children: React.ReactNode;
  authority: number;
  layoutClass?: string;
}

export const AppLayout: React.FC<Props> = (props) => {
  const { authority, layoutClass = "" } = props;
  const [offset, setOffset] = useState(0); // Track the scroll offset

  useZoomAccess();

  useEffect(() => {
    const handleScroll = () => {
      // TODO: Refactor this.
      setOffset(window.scrollY);
    };
    document.addEventListener("scroll", handleScroll, { passive: true });

    // Cleanup on unmount
    return () => {
      document.removeEventListener("scroll", handleScroll);
    };
  }, []);

  return (
    <RouteHandler authority={authority}>
      <Layout className={`App main-wrapper ${layoutClass}`}>
        <Layout.Header
          className={`main-header ${offset < 70 ? "transparent" : ""}`}
        >
          <Header />
        </Layout.Header>

        <Layout.Content>{props.children}</Layout.Content>

        <Layout.Footer>
          <FooterBlock />
        </Layout.Footer>
      </Layout>
    </RouteHandler>
  );
};
```

Technical Details:

**Props Interface:**
- `children: React.ReactNode` - Page content to render
- `authority: number` - Required user role (UserRoles enum)
- `layoutClass?: string` - Optional CSS class for page-specific styling

**State:**
- `offset: number` - Current scroll position (window.scrollY)

**Hooks:**
- `useZoomAccess()` - Custom hook pentru Zoom meetings access
- `useEffect` - Scroll event listener cu cleanup

**Scroll Tracking:**
- addEventListener("scroll") cu { passive: true } pentru performance
- Header becomes transparent când offset < 70px
- Cleanup removeEventListener în useEffect return

**Structure:**
1. RouteHandler wrapper (authorization)
2. Layout (Ant Design)
   - Layout.Header (sticky header cu transparency)
   - Layout.Content (page content)
   - Layout.Footer (footer block)

```tsx
// web/src/components/global/Header/index.tsx
import { Row, Col } from "antd/lib";
import React, { useState } from "react";
import Link from "next/link";
import { useDispatch, useSelector } from "react-redux";
import Image from "next/image";

import MainMenu from "./MainMenu/MainMenu";
import UserMenu from "./UserMenu/UserMenu";

import { PATHS } from "../../../../config/constants";
import logoWide from "@/assets/logo-normal-white.svg";
import logoWideBlack from "@/assets/logo-normal-black.svg";
import { getNextMeetingUrl } from "@/services/zoom.service";
import { UserRoles } from "@/@types/enums/user-role.enum";
import { AxiosResponse } from "axios";
import { createSubscription } from "@/services/api/Subscription";
import {
  CREATE_SUBSCRIPTION,
  SELECT_SUB_TYPE,
} from "@/redux/Actions/Subscription";
import { useRouter } from "next/router";
import { useZoomAccess } from "@/hooks/use-zoom-access";

interface HeaderProps {}

export const Header: React.FC<HeaderProps> = (): JSX.Element => {
  const authUser = useSelector((state: any) => state.user);
  const zoomAccess = useSelector((state: any) => state.zoomAccess);
  const [liveMeetingUrl, setLiveMeetingUrl] = React.useState<string>();
  const [isLoading, setIsLoading] = useState(false);
  const dispatch = useDispatch();
  const router = useRouter();

  const lpTABU = "/tabu";
  const lpSHS = "/somatichealingsummit";
  const lpPath = window.location.pathname;
  const isCustomLP = [lpTABU, lpSHS].includes(lpPath);

  const { subscriptions } = useZoomAccess();

  const subscriptionTypes = subscriptions?.map(
    (s: { subTypeId: any }) => s.subTypeId,
  );

  React.useEffect(() => {
    const fetchLiveMeetingUrl = async () => {
      if (
        authUser?.role > UserRoles.GUEST &&
        zoomAccess &&
        (zoomAccess.meetings || zoomAccess.webinars)
      ) {
        try {
          const url = await getNextMeetingUrl(
            zoomAccess.meetings,
            zoomAccess.webinars,
            subscriptionTypes,
          );
          if (url) {
            setLiveMeetingUrl(url);
          }
        } catch (error) {
          console.error("Error fetching live meeting URL:", error);
        }
      } else {
        setLiveMeetingUrl(undefined);
      }
    };

    const timeoutId = setTimeout(fetchLiveMeetingUrl, 300);

    return () => clearTimeout(timeoutId);
  }, [authUser?.role, zoomAccess]);

  const handleClick = async () => {
    if (isLoading) return;

    const subscriptionTypeId = 28;
    setIsLoading(true);

    try {
      if (Boolean(authUser?.id)) {
        try {
          const res: AxiosResponse =
            await createSubscription(subscriptionTypeId);

          if (res?.data?.id) {
            const subscription = res.data;

            // Dispatch the subscription to Redux store
            dispatch({
              type: CREATE_SUBSCRIPTION,
              items: [subscription],
            });

            // Redirect to dashboard after a short delay
            setTimeout(() => {
              router.push(PATHS.DASHBOARD);
            }, 500);
          }
        } catch (error) {
          console.error("Error creating subscription:", error);
          setIsLoading(false);
        }
      } else {
        dispatch({
          type: SELECT_SUB_TYPE,
          selectedSubscriptionTypeId: subscriptionTypeId,
        });
        await router.push(PATHS.SIGNUP);
      }
    } catch (error) {
      console.error("Error creating subscription:", error);
      setIsLoading(false);
    }
  };

  return (
    <div className="container">
      <Row>
        <Col xs={10} sm={8} className="logo-container">
          <Link href={isCustomLP ? lpPath : PATHS.HOME}>
            <Image
              src={logoWide}
              className="logo-navbar bg-transparent"
              alt="Somaway"
              width={140}
              height={42}
            />
            <Image
              src={logoWideBlack}
              className="logo-navbar bg-white"
              alt="Somaway"
              width={140}
              height={42}
            />
          </Link>
        </Col>
        <Col
          xs={14}
          sm={16}
          className={`menu-container menu-align-right ${
            authUser?.role ? "loggedin-menu" : ""
          }`}
        >
          {liveMeetingUrl &&
            authUser?.role &&
            zoomAccess &&
            (zoomAccess.meetings || zoomAccess.webinars) && (
              <a
                className="hardCTA pulseCTA"
                href={liveMeetingUrl || ""}
                style={{
                  backgroundColor: "#c7cd2d",
                  padding: "16px",
                  color: "#263f1b",
                  borderRadius: "16px",
                  fontWeight: "700",
                  marginRight: "10px",
                }}
                target="_blank"
                rel="noopener noreferrer"
              >
                LIVE
              </a>
            )}
          <MainMenu
            authUserRole={authUser?.role}
            liveMeetingUrl={liveMeetingUrl}
            onSignupClick={handleClick}
          />
          {authUser && authUser.role && <UserMenu authUser={authUser} />}
        </Col>
      </Row>
    </div>
  );
};
```

```vue
import { ref, computed, onMounted, watch } from 'vue'
import { useRouter, useRoute } from '#app'

import { useUserStore } from '~/stores/user'
import { useSubscriptionStore } from '~/stores/subscription'
import { useZoomStore } from '~/stores/zoom'
import { getNextMeetingUrl } from '~/services/zoom.service'
import { createSubscription } from '~/services/api/subscription'
import { UserRoles } from '~/types/enums/user-role.enum'
import { PATHS } from '~/config/constants'
import logoWide from '~/assets/logo-normal-white.svg'
import logoWideBlack from '~/assets/logo-normal-black.svg'

const router = useRouter()
const route = useRoute()
const userStore = useUserStore()
const subscriptionStore = useSubscriptionStore()
const zoomStore = useZoomStore()

const { user: authUser } = storeToRefs(userStore)
const { zoomAccess } = storeToRefs(zoomStore)
const { subscriptions } = storeToRefs(subscriptionStore)

const liveMeetingUrl = ref<string>()
const isLoading = ref(false)

const lpTABU = '/tabu'
const lpSHS = '/somatichealingsummit'
const lpPath = computed(() => route.path)
const isCustomLP = computed(() => [lpTABU, lpSHS].includes(lpPath.value))

const subscriptionTypes = computed(() =>
  subscriptions.value?.map((s: any) => s.subTypeId)
)

const showLiveButton = computed(() =>
  liveMeetingUrl.value &&
  authUser.value?.role &&
  zoomAccess.value &&
  (zoomAccess.value.meetings || zoomAccess.value.webinars)
)

const fetchLiveMeetingUrl = async () => {
  if (
    authUser.value?.role > UserRoles.GUEST &&
    zoomAccess.value &&
    (zoomAccess.value.meetings || zoomAccess.value.webinars)
  ) {
    try {
      const url = await getNextMeetingUrl(
        zoomAccess.value.meetings,
        zoomAccess.value.webinars,
        subscriptionTypes.value
      )
      if (url) {
        liveMeetingUrl.value = url
      }
    } catch (error) {
      console.error('Error fetching live meeting URL:', error)
    }
  } else {
    liveMeetingUrl.value = undefined
  }
}

const handleSignupClick = async () => {
  if (isLoading.value) return

  const subscriptionTypeId = 28
  isLoading.value = true

  try {
    if (authUser.value?.id) {

        const res = await createSubscription(subscriptionTypeId)

        if (res?.data?.id) {

    console.error('Error creating subscription:', error)
    isLoading.value = false
```

```tsx
// web/src/components/global/Button/button.tsx
import React from "react";
import { smoothScroll } from "@/lib/scrollUtils";

type ButtonProps = {
  onClick?: () => void;
  href?: string;
  size?: "small" | "medium" | "big";
  type?: "primary" | "secondary" | "fourth" | "link" | "tag" | "default";
  arrow?: "left" | "right";
  disabled?: boolean;
  className?: string;
  children: React.ReactNode;
  htmlType?: "button" | "submit" | "reset";
  loading?:
    | boolean
    | {
        delay?: number;
      };
  offset?: number;
  openInNewTab?: boolean;
};

const CustomButton: React.FC<ButtonProps> = ({
  onClick,
  href,
  size = "medium",
  type = "primary",
  arrow,
  disabled = false,
  className = "",
  children,
  htmlType = "button",
  offset = 0,
  openInNewTab = false,
}) => {
  const classNames =
    `custom-button custom-button-${size} custom-button-${type} ${
      disabled ? "disabled" : ""
    } ${arrow ? `custom-button-arrow-${arrow}` : ""} ${className}`.trim();

  const handleClick = (
    event: React.MouseEvent<HTMLButtonElement | HTMLAnchorElement>,
  ) => {
    if (disabled) {
      event.preventDefault();
      return;
    }

    if (href && href.startsWith("#")) {
      event.preventDefault();
      const sectionId = href.slice(1);
      const sectionElement = document.getElementById(sectionId);

      if (sectionElement) {
        const yPosition =
          sectionElement.getBoundingClientRect().top + window.scrollY;
        smoothScroll(yPosition, 500, offset);
      }
    } else if (!href && onClick) {
      onClick();
    }
  };

  const buttonContent = (
    <>
      {arrow === "left" && <span className="arrow left-arrow"></span>}
      {children}
      {arrow === "right" && <span className="arrow right-arrow"></span>}
    </>
  );

  if (href) {
    return (
      <a
        href={href}
        className={classNames}
        onClick={handleClick}
        role="button"
        aria-disabled={disabled}
        target={openInNewTab ? "_blank" : undefined}
        rel={openInNewTab ? "noopener noreferrer" : undefined}
      >
        {buttonContent}
      </a>
    );
  }

  return (
    <button
      type={htmlType}
      onClick={handleClick}
      className={classNames}
      disabled={disabled}
    >
      {buttonContent}
    </button>
  );
};

export default CustomButton;
```

- `arrow?: "left" | "right"` - Arrow decoration position
- `disabled?: boolean` - Disabled state

- `children: React.ReactNode` - Button content
- `htmlType?: "button" | "submit" | "reset"` - HTML button type

```vue
  }
```

```tsx
import { AutoComplete, Input, Typography } from 'antd/lib'
import { SearchOutlined } from '@ant-design/icons/lib'

import React, { useRef, useState, useEffect } from 'react'
import classNames from 'classnames'
import { useRouter } from 'next/router'

import { searchCourse } from '@/services/api/api'
import { DefaultOptionType } from 'antd/es/select'

const Search: React.FC = (): JSX.Element => {
	const inputEl = useRef<typeof Input>(null)
	const router = useRouter()

	const [value, setValue] = useState('')
	const [searchMode, setSearchMode] = useState(false)
	const [autoCompleteData, setAutoCompleteData] = useState<DefaultOptionType[]>([])
	const [noDataMessage, setNoDataMessage] = useState('')

	const enterSearchMode = (): void => {
		if (autoCompleteData) return
		setSearchMode(true)
		if (inputEl.current) {
			(inputEl.current as any).focus()
		}
	}

	const leaveSearchMode = (): void => {
		setValue('')
		setSearchMode(false)
	}

	useEffect(() => {
		if (value === '') setAutoCompleteData([])
		if (value.length < 3) setNoDataMessage('Enter at least 3 characters to search')
		else {
			searchCourse(value).then(res => {
				const foundTitles = res.data.titles.map((course: { slug: any; name: string }) => {
					return {
						value: course.slug,
						label: <Typography.Text style={{ fontSize: 15.7, width: '100%' }}> {course.name} </Typography.Text>
					}
				})
				if (foundTitles.length === 0) setAutoCompleteData([])
				else setAutoCompleteData(foundTitles)
				setNoDataMessage('Could not match anything')
			}).catch(_error => { })
		}
	}, [value])

	const handleOnSelect = (courseSlug: string): Promise<boolean> => router.push('/[course]', `/${courseSlug}`, { shallow: false })

	const inputClass = classNames('input', {
		'show': searchMode,
		'headerSearch': true,
		'search': true,
		'action': true,
	})

	return (
		<div className={inputClass} onClick={enterSearchMode}>
			<AutoComplete
				defaultActiveFirstOption
				notFoundContent={
					<Typography.Text strong style={{ width: '100%' }}>{noDataMessage}</Typography.Text>
				}
				className={inputClass}
				value={value}
				onChange={(value: string): void => setValue(value)}
				onSelect={handleOnSelect}
				onBlur={leaveSearchMode}
				options={autoCompleteData}
			>
				<Input
					ref={inputEl as any} placeholder="Cauta cursuri ..."
					onBlur={leaveSearchMode}
				/>
			</AutoComplete>
			<SearchOutlined
```

2. Session validation: getCachedToken() + handleSessionExpiration()
3. Loading state: setTimeout 1000ms before showing content
4. Error states: 403 (not logged in), 404 (insufficient authority)
5. Result component: Ant Design Result cu status + title + buttons
6. Buttons: "Inapoi la login" (403) sau "Back Home" (404)
7. Page loaded: div className="page-loaded" wrapper pentru content

Funcționalitate:
- Authority check: if pageAuthority > UserRoles.GUEST
- Session validation: validate token din cache
- Loading delay: 1000ms setTimeout before page render
- 403 Error: user not logged in → redirect to login
- 404 Error: user logged in but insufficient authority
- Children: render page content if authorized

Este folosită în AppLayout (wraps all protected pages).

```tsx
// web/src/middleware/RouteHandler.tsx
import { Button, Result } from 'antd/lib'

import React, { useState, useEffect } from 'react'
import { useSelector } from 'react-redux'
import { useRouter } from 'next/router'

import { getCachedToken, handleSessionExpiration } from '@/services/auth.service'
import { UserRoles } from '@/@types/enums/user-role.enum'
import { NOT_INITIALIZED, PATHS } from '../../config/constants'
import LoadingPage from '@/components/global/LoadingPage/LoadingPage'

export const RouteHandler: React.FC<{ authority: number, children: any }> = ({ authority: pageAuthority, children }) => {
	const router = useRouter()
	const [isLoading, setIsLoading] = useState(true);
	const authUser = useSelector((state: any) => state.user)

	const handleInitialProcessesOnRoute = async (): Promise<void> => {
		const authToken = getCachedToken('atx')
		if (!authUser?.id || authToken === NOT_INITIALIZED) {
			await handleSessionExpiration()
		}
	}

	useEffect(() => {
		if (pageAuthority > UserRoles.GUEST) {
			handleInitialProcessesOnRoute()
		}
		setTimeout(() => {
			setIsLoading(false)
		}, 1000)
	}, [])

	if (isLoading) {
		return <LoadingPage />
	}

	// if page pageAuthority is higher than guest and user is not logged in
	if (pageAuthority > UserRoles.GUEST && !authUser) {
		return (<Result
			status="403"
			title="403"
			subTitle="Probleme cu accesul la pagina"
			extra={
				<Button type="primary" onClick={() => router.push(PATHS.SIGN_INL)}>Inapoi la login</Button>
			}
		/>
		)
	}
	//if user is logged in and has lower authority than the page pageAuthority
	else if (authUser && pageAuthority > Object.values(UserRoles).indexOf(authUser.role)) {
		return (
			<Result
				status="404"
				title="404"
				subTitle="Aceasta pagina nu exista"
				extra={
					<Button type="primary" onClick={() => router.push('/')}>Back Home</Button>
				}
			/>
		)
	}

	return <div className="page-loaded"> {children}</div>
}
```

```typescript
// web/middleware/routeHandler.ts
import { defineNuxtRouteMiddleware, navigateTo } from '#app'
import { useUserStore } from '~/stores/user'
import { useRouter } from '#app'
import { getCachedToken, handleSessionExpiration } from '~/services/auth.service'
import { UserRoles } from '~/types/enums/user-role.enum'
import { NOT_INITIALIZED, PATHS } from '~/config/constants'

export default defineNuxtRouteMiddleware(async (to, from) => {
  const userStore = useUserStore()
  const router = useRouter()

  // Get page authority from route meta
  const pageAuthority = (to.meta.authority as number) || UserRoles.GUEST

  // Session validation for protected routes
  if (pageAuthority > UserRoles.GUEST) {
    const authToken = getCachedToken('atx')
    if (!userStore.user?.id || authToken === NOT_INITIALIZED) {
      await handleSessionExpiration()
    }
  }

  // Check authorization
  if (pageAuthority > UserRoles.GUEST && !userStore.user) {
    // Not logged in - redirect to login
    return navigateTo(PATHS.SIGN_INL)
  }

  if (userStore.user && pageAuthority > Object.values(UserRoles).indexOf(userStore.user.role)) {
    // Insufficient authority - redirect to home
    return navigateTo('/')
  }
})
```
